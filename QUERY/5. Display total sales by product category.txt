db.getCollection('Past Order').aggregate(
  [
    {
      $group: {
        _id: '$order_type',
        total_sales: { $sum: 1 },
        total_revenue: {
          $sum: { $toDouble: '$order_cost' }
        }
      }
    }
  ],
  { maxTimeMS: 60000, allowDiskUse: true }
);
